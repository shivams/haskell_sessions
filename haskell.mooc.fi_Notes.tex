\chapter{Lecture 1: And so It Begins}
\label{sec:org74b95ff}
\section{Haskell Characteristics}
\label{sec:org20221f6}
\begin{itemize}
\item Functional, pure, and lazy
\item Strongly typed with type inference
\item Garbage collected and compiled
\item \textbf{Tooling}: \texttt{stack ghci} (REPL), \texttt{:t} (type), \texttt{:load}, \texttt{:reload}
\end{itemize}
\section{Fundamental Concepts}
\label{sec:org33be1ca}
\begin{itemize}
\item \textbf{Expressions}: Everything is an expression (no statements)
\item \textbf{Application}: \texttt{f x y} (not \texttt{f(x,y)}); function calls bind tightest
\item \textbf{Basic Types}: \texttt{Int}, \texttt{Integer}, \texttt{Double}, \texttt{Bool}, \texttt{String} (\texttt{[Char]})
\item \textbf{Currying}: \texttt{a -> b -> c} represents multi-argument functions
\end{itemize}
\section{Program Structure}
\label{sec:orgfadcb47}
\begin{itemize}
\item \textbf{Modules}: One per file
\item \textbf{Definitions}: Optional type signature + equations
\item \textbf{Logic}: \texttt{if...then...else} (else is mandatory), \texttt{where=/=let}
\item \textbf{Immutability}: Bindings are constant; ``updates'' create new values
\end{itemize}
\chapter{Lecture 2: Recursion \& Lists}
\label{sec:org12df93a}
\section{Recursion \& Control Flow}
\label{sec:orga5f6e25}
\begin{itemize}
\item \textbf{Helper Functions}: Often marked with \texttt{'} (e.g., \texttt{fibonacci'}) to manage state
\item \textbf{Tail Recursion}: Recursive call is the last action \(\rightarrow\) optimizes to a loop
\item \textbf{Guards}: \texttt{| condition = value} (first true guard wins)
\end{itemize}
\section{Lists \& Polymorphism}
\label{sec:org28ca8fe}
\begin{itemize}
\item \textbf{Lists}: Homogeneous \texttt{[a]}, singly-linked, constructed via cons \texttt{:}
\item \textbf{Common Ops}: \texttt{head}, \texttt{tail}, \texttt{take}, \texttt{drop}, \texttt{++}, \texttt{!!}, \texttt{reverse}
\item \textbf{Polymorphism}: Lowercase type variables (e.g., \texttt{a}) determined by unification
\end{itemize}
\section{Handling Optionality}
\label{sec:org2ab370f}
\begin{itemize}
\item \textbf{Maybe a}: \texttt{Nothing | Just x} (Safe alternative to null)
\item \textbf{Either a b}: \texttt{Left a | Right b} (Commonly used for success/error)
\item \textbf{Case expressions}: \texttt{case ... of} for complex pattern matching
\end{itemize}
\chapter{Lecture 3: Higher-Order Functions}
\label{sec:org0154a36}
\section{Functions as First-Class Citizens}
\label{sec:org732ac5d}
\begin{itemize}
\item \textbf{HOFs}: Functions that take or return functions (e.g., \texttt{map}, \texttt{filter})
\item \textbf{Partial Application}: \texttt{add 3} returns a function waiting for the second arg
\item \textbf{Sections}: Partially applied operators, e.g., \texttt{(+1)}, \texttt{(*2)}
\end{itemize}
\section{Composition \& Lambdas}
\label{sec:org4f0bbc5}
\begin{itemize}
\item \textbf{Composition}: \texttt{(f . g) x = f (g x)}
\item \textbf{Application Operator}: \texttt{\$} (low precedence, avoids parentheses)
\item \textbf{Lambdas}: Anonymous functions \texttt{\textbackslash{}x -> x + 1}
\end{itemize}
\section{Advanced List Patterns}
\label{sec:orgbec4f4c}
\begin{itemize}
\item \textbf{Comprehensions}: \texttt{[x | x <- xs, cond]}
\item \textbf{Recursion Patterns}: Building via \texttt{:} and consuming via \texttt{x:xs}
\item \textbf{Typed Holes}: Using \texttt{\_} to let GHC suggest missing types
\end{itemize}
\chapter{Lecture 4: Type Classes \& Folds}
\label{sec:org14190aa}
\section{Tuples \& Folds}
\label{sec:org496ed99}
\begin{itemize}
\item \textbf{Tuples}: Fixed-length, mixed types; accessed via \texttt{fst=/=snd}
\item \textbf{foldr}: ``Crunches'' a list into a single value
\begin{itemize}
\item \texttt{foldr f z [] = z}
\item \texttt{foldr f z (x:xs) = f x (foldr f z xs)}
\end{itemize}
\end{itemize}
\section{Type Classes}
\label{sec:org2394aad}
\begin{itemize}
\item \textbf{Definition}: Groups of types sharing common operations (not OOP classes)
\item \textbf{Constraints}: \texttt{(Eq a) => ...} restricts polymorphism to types that implement \texttt{Eq}
\item \textbf{Key Classes}:
\begin{itemize}
\item \texttt{Eq} (\(\approx\), \(\neq\)) and \texttt{Ord} (\(<\), \(>\), \texttt{compare})
\item \texttt{Num} \(\rightarrow\) \texttt{Integral} / \texttt{Fractional} \(\rightarrow\) \texttt{Floating}
\item \texttt{Show} (to string) and \texttt{Read} (from string)
\end{itemize}
\end{itemize}
\chapter{Lecture 5: Algebraic Datatypes (ADTs)}
\label{sec:org6586f71}
\section{Defining Custom Types}
\label{sec:org8716247}
\begin{itemize}
\item \textbf{ADTs}: Sum of constructors, product of fields (\texttt{data T = C1 | C2 Int String})
\item \textbf{Deriving}: Automatic implementation of \texttt{Show}, \texttt{Eq}, \texttt{Read}
\item \textbf{Type Parameters}: Generic types, e.g., \texttt{data Maybe a}
\end{itemize}
\section{Recursive Types \& Trees}
\label{sec:org527452b}
\begin{itemize}
\item \textbf{Recursive Types}: Types that refer to themselves (e.g., custom \texttt{List a})
\item \textbf{Trees}: \texttt{Node val left right | Empty}
\item \textbf{Operations}: Height calculation, BST lookup, and insertion
\end{itemize}
\section{Record Syntax}
\label{sec:org02b8590}
\begin{itemize}
\item Named fields instead of positional arguments
\item Automatically generates accessor functions
\item Improves readability and maintainability for large data structures
\end{itemize}
